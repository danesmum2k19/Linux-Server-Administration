IF(SUSER_ID('$DOMAIN\$USER') IS NULL)BEGIN CREATE LOGIN [$DOMAIN\$USER] FROM WINDOWS;/*WINDOWS_GROUP*/ END;

USE [$DATABASE_NAME]

IF NOT EXISTS (SELECT [name] FROM sys.database_principals WHERE [name] =  '$DOMAIN\$USER') BEGIN CREATE USER  [$DOMAIN\$USER] FOR LOGIN [$DOMAIN\$USER]  END; 

GRANT CONNECT TO [$DOMAIN\$USER]
GRANT EXECUTE ON [dbo].[$STORE_PROCEDURE] TO [$DOMAIN\$USER]
GO
